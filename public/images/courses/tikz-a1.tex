	\begin{scope}[

		\path[name path = b3] (B) -- ++(\b+2*\th{b}:8);
		
		\path[name path = a1] (A) -- ++(\a:8);
		\path[name path = a2] (A) -- ++(\a+\th{a}:8);
		\path[name path = a3] (A) -- ++(\a+2*\th{a}:8);
		
		\path[name intersections = {of = b2 and a2, by = C}];
		
		\path[name intersections = {of = b3 and a3, by = a}];
		\path[name intersections = {of = b1 and a1, by = b}];
		
		% angles
		\pic[
			"$\alpha$" black,
			draw, semithick, gray, fill = gray!10,
			angle radius = 4.5mm, angle eccentricity = 1.5
		]
			{angle = b--A--C};
		\pic[
			"$\alpha$" black,
			draw, semithick, gray, fill = gray!10,
			angle radius = 4.5mm, angle eccentricity = 1.5
		]
			{angle = C--A--a};
			
		\pic[
			"$\beta$" black,
			draw, semithick, gray, fill = gray!10,
			angle radius = 5.5mm, angle eccentricity = 1.4
		]
			{angle = b--B--C};
		\pic[
			"$\beta$" black,
			draw, semithick, gray, fill = gray!10,
			angle radius = 5.5mm, angle eccentricity = 1.4
		]
			{angle = C--B--a};

			"$\gamma$" black,
			draw, semithick, gray, fill = gray!10,
			angle radius = 4.5mm, angle eccentricity = 1.5

			{angle = A--C--B};

			"$\theta$" black,

		% labels

		% points
